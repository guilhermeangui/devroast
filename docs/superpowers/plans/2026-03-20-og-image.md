# OG Image for Roast Results Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Generate a dynamic OpenGraph image for each roast result page using Takumi, so shared links embed a rich preview with the score, verdict, language, and roast quote.

**Architecture:** Next.js App Router's `opengraph-image.tsx` file convention auto-wires the `og:image` metadata tag. A default-export async function in that file fetches the roast from Postgres via Drizzle and returns a 1200×630 PNG rendered by Takumi's Rust-backed `ImageResponse`. The existing `generateMetadata` in `page.tsx` is enriched with `openGraph` and `twitter` fields (the image URL itself is injected automatically by Next.js from the sibling file).

**Tech Stack:** `@takumi-rs/image-response`, Drizzle ORM, Next.js 16 App Router (Node.js runtime)

### Task 2: Create the `opengraph-image.tsx` route

**Files:**
- Create: `src/app/roast/[id]/opengraph-image.tsx`

This file is auto-detected by Next.js App Router as the OG image for `/roast/[id]`. It must use a **default export async function** (not a named `GET` export — that is for `route.ts` files). The named exports `size`, `contentType`, and `alt` configure the image metadata.

The visual design (from the Pencil frame `4J5QT`) is:
- Background: `#0a0a0a`
- Top center: `> devroast` in `#22c55e` (green), monospace, `14px`
- Center: score number in `#f59e0b` (amber), `120px`, bold + `/10` in `#4b5563` (gray), `40px`
- Verdict row: colored dot (8px circle) + verdict text in monospace, `16px`
- Meta row: `lang: {language} · {lineCount} lines` in `#6b7280`, `14px`
- Quote row: `"{roastQuote}"` in `#9ca3af`, `16px`, italic, monospace, max-width `800px`, text-centered

Verdict dot color map:
```
needs_serious_help  → #ef4444  (red)
pretty_bad          → #f97316  (orange)
could_be_worse      → #eab308  (yellow)
not_terrible        → #3b82f6  (blue)
surprisingly_good   → #22c55e  (green)
```

- [ ] **Step 1: Create the file**

```tsx
// src/app/roast/[id]/opengraph-image.tsx
import { ImageResponse } from "@takumi-rs/image-response";
import { db } from "@/db";
import { roasts } from "@/db/schema";
import { eq } from "drizzle-orm";

export const runtime = "nodejs";

export const size = { width: 1200, height: 630 };
export const contentType = "image/png";
export const alt = "devroast score card";

const verdictColor: Record<string, string> = {
  needs_serious_help: "#ef4444",
  pretty_bad: "#f97316",
  could_be_worse: "#eab308",
  not_terrible: "#3b82f6",
  surprisingly_good: "#22c55e",
};

export default async function Image({
  params,
}: {
  params: Promise<{ id: string }>;
}) {
  const { id } = await params;

  const [roast] = await db
    .select({
      score: roasts.score,
      verdict: roasts.verdict,
      language: roasts.language,
      lineCount: roasts.lineCount,
      roastQuote: roasts.roastQuote,
    })
    .from(roasts)
    .where(eq(roasts.id, id))
    .limit(1);

  if (!roast) {
    return new ImageResponse(
      <div
        style={{
          display: "flex",
          width: "100%",
          height: "100%",
          background: "#0a0a0a",
          alignItems: "center",
          justifyContent: "center",
        }}
      >
        <span style={{ color: "#22c55e", fontFamily: "monospace", fontSize: 24 }}>
          {">"} devroast
        </span>
      </div>,
      {
        ...size,
        headers: { "Cache-Control": "public, immutable, max-age=31536000" },
      },
    );
  }

  const dotColor = verdictColor[roast.verdict] ?? "#6b7280";
  const [whole, decimal] = Number(roast.score).toFixed(1).split(".");

  return new ImageResponse(
    <div
      style={{
        display: "flex",
        flexDirection: "column",
        width: "100%",
        height: "100%",
        background: "#0a0a0a",
        alignItems: "center",
        justifyContent: "center",
        gap: 16,
        padding: "0 80px",
      }}
    >
      {/* Branding */}
      <span
        style={{
          color: "#22c55e",
          fontFamily: "monospace",
          fontSize: 14,
          letterSpacing: 1,
        }}
      >
        {">"} devroast
      </span>

      {/* Score */}
      <div style={{ display: "flex", alignItems: "flex-end", lineHeight: 1 }}>
        <span
          style={{
            color: "#f59e0b",
            fontSize: 120,
            fontWeight: 700,
            lineHeight: 1,
          }}
        >
          {whole}.{decimal}
        </span>
        <span
          style={{
            color: "#4b5563",
            fontSize: 40,
            fontWeight: 400,
            paddingBottom: 16,
            marginLeft: 4,
          }}
        >
          /10
        </span>
      </div>

      {/* Verdict */}
      <div style={{ display: "flex", alignItems: "center", gap: 8 }}>
        <div
          style={{
            width: 8,
            height: 8,
            borderRadius: 4,
            background: dotColor,
          }}
        />
        <span
          style={{
            color: dotColor,
            fontFamily: "monospace",
            fontSize: 16,
          }}
        >
          {roast.verdict}
        </span>
      </div>

      {/* Language + lines */}
      <span style={{ color: "#6b7280", fontFamily: "monospace", fontSize: 14 }}>
        lang: {roast.language} · {roast.lineCount} lines
      </span>

      {/* Roast quote */}
      <span
        style={{
          color: "#9ca3af",
          fontFamily: "monospace",
          fontSize: 16,
          fontStyle: "italic",
          textAlign: "center",
          maxWidth: 800,
        }}
      >
        "{roast.roastQuote}"
      </span>
    </div>,
    {
      ...size,
      headers: { "Cache-Control": "public, immutable, max-age=31536000" },
    },
  );
}
```

```ts
export async function generateMetadata({ params }: Props): Promise<Metadata> {
  const { id } = await params;

  const [roast] = await db
    .select({
      score: roasts.score,
      verdict: roasts.verdict,
      roastQuote: roasts.roastQuote,
    })
    .from(roasts)
    .where(eq(roasts.id, id))
    .limit(1);

  if (!roast) {
    return { title: "Roast Result | devroast" };
  }

  const title = `${roast.score}/10 — ${roast.verdict} | devroast`;
  const description = `"${roast.roastQuote}"`;

  return {
    title,
    description,
    openGraph: {
      title,
      description,
      type: "website",
    },
    twitter: {
      card: "summary_large_image",
      title,
      description,
    },
  };
}
```
